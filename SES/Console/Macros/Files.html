def self.append_file(file, contents)
  assign_last(file)
  begin
    File.open(@last, 'a') { |f| f.write(contents) }
    File.read(@last)
  rescue
    nil
  end
end

def self.assign_last(*files)
  file = files.reject { |file| file.empty? }.last
  @last = file unless file.nil?
  @last
end

def self.delete(*files)
  assign_last(*files)
  File.delete(*(files.empty? ? [@last] : files))
end
